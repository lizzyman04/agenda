```dart
import 'package:injectable/injectable.dart';

/// DI registrations for the Task domain.
///
/// ItemRepositoryImpl and RecurrenceEngineImpl are registered via their
/// @LazySingleton(as: ...) annotations — injectable_generator picks them up.
/// This module registers ItemDao and ItemMapper which have no annotations.
@module

  ///
  /// Pure converter — no dependencies.
  @lazySingleton
  ItemMapper get itemMapper => const ItemMapper();
}
```

```dart
// In item_repository.dart:
AsyncResult<List<String>> getDistinctGtdContexts();

// In item_repository_impl.dart:
@override
AsyncResult<List<String>> getDistinctGtdContexts() async {
  try {
    final allResult = await filterItems(showCompleted: false);
    if (allResult is Err) return allResult as Err<List<String>>;
    final items = (allResult as Success<List<Item>>).value;
    final contexts = items
        .map((i) => i.gtdContext)
        .whereType<String>()
        .toSet()
        .toList()
      ..sort();
    return Success(contexts);
  } catch (e) {
    return Err(DatabaseFailure('getDistinctGtdContexts failed: $e'));
  }
}
```

**Recurring task completion in TaskListCubit:** The `completeItem` method already exists. Extend it to handle recurring task creation. Add `RecurrenceEngine` as a constructor dependency of `TaskListCubit`:

```dart
// In task_list_cubit.dart — updated constructor:
@injectable
class TaskListCubit extends Cubit<TaskListState> {
  TaskListCubit(this._repository, this._recurrenceEngine)
      : super(const TaskListInitial());

  final ItemRepository _repository;
  final RecurrenceEngine _recurrenceEngine;
  // ... existing fields ...

  Future<void> completeItem(Item item) async {
    final now = DateTime.now();
    final updated = item.copyWith(
      isCompleted: true,
      completedAt: now,
      updatedAt: now,
    );
    final updateResult = await _repository.updateItem(updated);
    if (updateResult is Err) {
      emit(TaskListError((updateResult as Err).failure));
      return;
    }

    // Recurring task: create next occurrence (TASK-10)
    if (item.recurrenceRule != null && item.dueDate != null) {
      final rule = _recurrenceEngine.parse(item.recurrenceRule);
      if (rule != null) {
        final nextDate = _recurrenceEngine.nextOccurrence(item.dueDate!, rule);
        final nextItem = item.copyWith(
          id: 0,  // new Isar autoincrement
          isCompleted: false,
          completedAt: null,
          deletedAt: null,
          dueDate: nextDate,
          createdAt: now,
          updatedAt: now,
        );
        await _repository.createItem(nextItem);
        // watchChanges() stream fires; _reload() is called by listener
      }
    }
    // watchChanges() fires reload
  }
}
```

**Note on copyWith and id=0:** The `Item.copyWith` method uses `id ?? this.id` — to create a new item with `id=0`, pass `id: 0` explicitly. This is correct because Isar treats `id=0` as a new item (autoincrement assigns a new id). If `copyWith` does not support setting id to 0 (because `0 ?? this.id` would use `this.id` for `null` check — actually `0 is not null` so `0 ?? this.id == 0`), this is safe.

## Threat Model

| Threat ID | Description | Severity | Disposition | Mitigation |
|-----------|-------------|----------|-------------|------------|
| T-02-01 | Isar query injection via user search string | medium | mitigate | SearchBar onChanged calls cubit.search() which delegates to repository.searchByTitle() — typed Isar API used throughout |
| T-02-02 | Unbounded Isar result sets cause OOM | medium | mitigate | getDistinctGtdContexts() uses filterItems() which has .limit(500) — large context count is impossible in practice |
| T-02-04 | parent_id cycle (subtask pointing to subtask) | medium | mitigate | ItemRepositoryImpl.createItem validates parentId type before write — covered in repository impl tests |
| T-02-05 | Soft-deleted items visible via stale Cubit state | medium | mitigate | watchChanges() triggers full _reload() after every delete/restore; GTD filter screen reads from cubit's live state |
